# Example cron job to run GitDigest daily at 9 AM
# Edit with: crontab -e
# Add this line:

# Option 1: Direct run (simpler)
0 9 * * * cd /path/to/GitDigest && GITHUB_TOKEN=your_token /usr/bin/python3 run_digest.py >> output/cron.log 2>&1

# Option 2: Using automation wrapper (recommended for production)
0 9 * * * GITHUB_TOKEN=your_token /path/to/GitDigest/automation/daily-gitdigest.sh

# Option 3: With environment file
0 9 * * * /bin/bash -c "source ~/.bashrc && cd /path/to/GitDigest && python3 run_digest.py >> output/cron.log 2>&1"

# To see current cron jobs: crontab -l
# To remove all cron jobs: crontab -r

# Troubleshooting tips:
# - Use full paths for all commands
# - Set GITHUB_TOKEN in the cron line or source it from ~/.bashrc
# - Check cron logs: tail -f /var/log/cron (Linux) or check Console.app (macOS)
# - Test your command manually first
